void BubbleSort(int a[], int n)
{
	int i, j, t;
	for (i = 1; i < n; i++)
		for (j = n - 1; j >= i; j--)
			if (a[j - 1] > a[j]) // сравнение двух соседних элементов 
			{
				t = a[j - 1];
				a[j - 1] = a[j]; //перестановка элементов
				a[j] = t;
			}
}

void InsertSort(int m[], int n)
{
	int t, i, j;
	for (i = 1; i < n; i++)
	{
		t = m[i];
		for (j = i - 1; j >= 0 && m[j] > t; j--) // вложенный цикл ищем место для вставки
			m[j + 1] = m[j]; m[j + 1] = t; //элементы меняются местами
	}
}
